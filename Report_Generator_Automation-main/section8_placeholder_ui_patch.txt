def load_section8_placeholder_data(self):
    """Load Section 8 placeholder data from configuration"""
    try:
        if hasattr(self, 'section8_placeholder_ui'):
            # Load placeholder data from config
            placeholder_data = self.config.get('section8_placeholders', [])
            
            # Update the UI with loaded data
            self.section8_placeholder_ui.load_placeholder_data(placeholder_data)
            
            print("[OK] Section 8 placeholder data loaded successfully")
            
    except Exception as e:
        print(f"[ERROR] Failed to load Section 8 placeholder data: {e}")

# 5. Add this to the save_configuration method (to save placeholder data)
def save_section8_placeholder_data(self):
    """Save Section 8 placeholder data to configuration"""
    try:
        if hasattr(self, 'section8_placeholder_ui'):
            # Get placeholder data from UI
            placeholder_data = self.section8_placeholder_ui.get_placeholder_data()
            
            # Save to config
            self.config['section8_placeholders'] = placeholder_data
            
            print("[OK] Section 8 placeholder data saved successfully")
            
    except Exception as e:
        print(f"[ERROR] Failed to save Section 8 placeholder data: {e}")
